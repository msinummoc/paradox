
#判断能否修建200的贸易公司建筑
can_inverstment_trade_company_cost_200 = {
	custom_trigger_tooltip = {
		tooltip = CAN_INVERSTMENT_TRADE_COMPANY_COST_200
		is_owned_by_trade_company = yes
		OR = {
			AND = {
				NOT = {
					has_trade_company_investment_in_area = {
						investment = local_quarter
						investor = ROOT
					}
				}
				NOT = {
					has_trade_company_investment_in_area = {
						investment = permanent_quarters
						investor = ROOT
					}
				}
			}
			AND = {
				NOT = {
					has_trade_company_investment_in_area = {
						investment = company_warehouse
						investor = ROOT
					}
				}
				NOT = {
					has_trade_company_investment_in_area = {
						investment = company_depot
						investor = ROOT
					}
				}
			}
			AND = {
				NOT = {
					has_trade_company_investment_in_area = {
						investment = brokers_office
						investor = ROOT
					}
				}
				NOT = {
					has_trade_company_investment_in_area = {
						investment = brokers_exchange
						investor = ROOT
					}
				}
			}
			AND = {
				NOT = {
					has_trade_company_investment_in_area = {
						investment = settlements
						investor = ROOT
					}
				}
				NOT = {
					has_trade_company_investment_in_area = {
						investment = district
						investor = ROOT
					}
				}
			}
			AND = {
				NOT = {
					has_trade_company_investment_in_area = {
						investment = company_administration
						investor = ROOT
					}
				}
				NOT = {
					has_trade_company_investment_in_area = {
						investment = military_administration
						investor = ROOT
					}
				}
			}
		}
	}
}

#判断能否修建1000的贸易公司建筑
can_inverstment_trade_company_cost_1000 = {
	custom_trigger_tooltip = {
		tooltip = CAN_INVERSTMENT_TRADE_COMPANY_COST_1000
		is_owned_by_trade_company = yes
		#1000的建筑只能修建一个，这里所有的1000建筑都不能存在才能修建
		NOT = {
			num_investments_in_trade_company_region = {
				investment = officers_mess
				value = 1
			}
		}
		NOT = {

				value = 1
			}
		}
		NOT = {
			num_investments_in_trade_company_region = {

				value = 1
			}

	}
