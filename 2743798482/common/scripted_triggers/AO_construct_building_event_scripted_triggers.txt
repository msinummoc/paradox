#判断能否修建士兵之家
can_build_soldier_households = {
	can_build = soldier_households
	is_state = yes
	OR = {
		trade_goods = grain #谷物
		trade_goods = wine	#酒类
		trade_goods = fish	#鱼类
		trade_goods = livestock	#牲畜
	}
	is_core = owner
}
#能否修建州议会
can_build_state_house = {
	can_build = state_house
	is_state = yes
	all_province_in_state = {
		is_capital = no
	}
	OR = {
		trade_goods = paper #纸张
		trade_goods = glass	#玻璃
		trade_goods = gems	#宝石
	}
	is_core = owner
}
#不合适的州议会
unsuitable_state_house = {
	OR = {
		has_building = state_house
		has_construction = state_house
	}
	OR = {
		#不生产 纸张玻璃宝石
		NOT = {
			trade_goods = paper #纸张
			trade_goods = glass	#玻璃
			trade_goods = gems	#宝石
		}
		#是首都直属州
		any_province_in_state = {
			is_capital = yes
		}
		#不是直属州
		is_state = no
	}
}
#不合适的士兵之家
unsuitable_soldier_households = {
	OR = {
		has_building = soldier_households
		has_construction = soldier_households
	}
	NOT = {
		trade_goods = grain #谷物
		trade_goods = wine	#酒类
		trade_goods = fish	#鱼类
		trade_goods = livestock	#牲畜
	}
}
#不合适的工厂
unsuitable_manufactory = {
	is_state = yes
	AND = {
		NOT = {
			trade_goods = paper #纸张
			trade_goods = glass	#玻璃
			trade_goods = gems	#宝石
		}
		all_province_in_state = {
			is_capital = no
		}
	}
	OR = {
		has_building = farm_estate
		has_building = wharf
		has_building = textile
		has_building = weapons
		has_building = plantations
		has_building = tradecompany
		has_building = mills

		has_construction = farm_estate
		has_construction = wharf
		has_construction = textile
		has_construction = weapons
		has_construction = plantations
		has_construction = tradecompany
		has_construction = mills
	}
	OR = {
		trade_goods = grain #谷物
		trade_goods = wine	#酒类
		trade_goods = fish	#鱼类
		trade_goods = livestock	#牲畜
	}
}
#判断能否修建普通工厂
can_build_manufactory = {
	OR = {
		can_build_manufactory_in_state = yes
		can_build_manufactory_in_company = yes
		can_build_manufactory_in_territory = yes
	}
	is_core = owner
}
#可以在直属州修工厂
can_build_manufactory_in_state = {
	is_state = yes
	OR = {
		can_build = farm_estate
		can_build = wharf
		can_build = textile
		can_build = weapons
		can_build = plantations
		can_build = tradecompany
		can_build = mills
		can_build = furnace
	}
	can_build_state_house = no
	can_build_soldier_households = no
	is_core = owner
}
#可以在贸易公司修工厂
can_build_manufactory_in_company = {
	is_owned_by_trade_company = yes
	OR = {
		can_build = farm_estate
		can_build = wharf
		can_build = textile
		can_build = weapons
		can_build = plantations
		can_build = tradecompany
		can_build = mills
		can_build = furnace
	}
	is_core = owner
}
#可以在自治领地修工厂
