# 降低省份自治度
AO_province_decrease_autonomy_effect = {
	add_local_autonomy = -25
	add_province_modifier = {
		name = autonomy_decreased
		duration = 10950 # 30 years
	}

	# 提高专制度
	export_to_variable = {
		which = absolutismVal
		value = development
		who = THIS
	}
	# 提高的专制度=发展度/20
	divide_variable = {
		which = absolutismVal
		value = 20
	}
	while = {
		limit = {
			check_variable = { 
				which = absolutismVal
				value = 0.05
			}
		}
		subtract_variable = { 
			which = absolutismVal 
			value = 0.05
		}
		owner = {
			add_absolutism = 0.05
		}
	}
}

# 增加省份自治度
