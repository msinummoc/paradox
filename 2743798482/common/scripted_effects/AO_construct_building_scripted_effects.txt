# 修建法院
AO_construct_courthouse_effect = {
	while = {
		limit = {
			treasury = 200
			any_owned_province = {
				# 24发展度 至少减6行政容量
				development = 24
				province_can_build_courthouse = yes
			}
		}
		random_owned_province = {
			limit = {
				development = 24

#在直属州修建工厂
AO_constract_manufactory_in_state = {
	if = {
		limit = {
			can_build_manufactory_in_state = yes

		}
		add_building_construction = {
			building = farm_estate
			speed = 1
			cost = 1
		}
	}
	else_if = {
