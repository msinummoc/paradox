country_event = {
	id = dismantle_building_events.1
	title = dismantle_building_events.1.title
    desc = dismantle_building_events.1.desc
    picture = BANKRUPTCY_eventPicture 

	trigger = {
		OR = {
			adm_tech = 15
			dip_tech = 15
			mil_tech = 15
		}
		any_owned_province = {
			NOT = {
				num_free_building_slots = 1
			}
		}
		ai = no
	}
	
	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = dismantle_building_events.1.cancel
	}

	option = {
		name = dismantle_building_events.1.dismantle
		country_event = {
			id = dismantle_building_events.2
		}
	}
}
