#如果判断当地传教力量小于0，自动撤回传教士
country_event = {
	id = auto_mission_events.2
	title = none
	desc = none
	picture = none
	hidden = yes

	trigger = {
		can_auto_mission = yes
		has_country_flag = AutoMission
		any_owned_province = {
			has_missionary = yes		#本地有传教士
			never_finish_the_mission = {
				total_missionary_strength = 0
			}
		}
	}

	mean_time_to_happen = {
		days = 30
	}
	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_missionary = yes		#本地有传教士
					never_finish_the_mission = {
						total_missionary_strength = 0
					}
				}
				#停止传教
				cancel_construction = missionary
			}
		}
	}
	option = {
	}	
}
